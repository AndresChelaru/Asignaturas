CREATE TRIGGER actualizar_importe
AFTER INSERT
ON lineapedido FOR EACH ROW
BEGIN
	UPDATE pedidos
	SET importe = importe + NEW.cantidad * NEW.precio
	WHERE numeroPedido = NEW.numeroPedido
END IF ;
